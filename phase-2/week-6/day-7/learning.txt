1) 🪝 Custom Hook → useCart
function useCart() {
  const [cart, setCart] = useState([]);

  const add = useCallback((product) => {
    setCart((prev) => [...prev, product]);
  }, []);

  const remove = useCallback((id) => {
    setCart((prev) => prev.filter((x) => x.id !== id));
  }, []);

  return { cart, add, remove };
}
