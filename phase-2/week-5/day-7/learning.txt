1) 🔍 SearchBar
function SearchBar({ query, setQuery }) {
  return (
    <input
      value={query}
      onChange={(e) => setQuery(e.target.value)}
      placeholder="Search posts..."
    />
  );
}
2) ➕ Add Post Form
function AddPostForm({ onAdd }) {
  const [title, setTitle] = useState("");
  const [author, setAuthor] = useState("");

  function handleSubmit(e) {
    e.preventDefault();

    onAdd({
      id: Date.now(),
      title,
      author,
      likes: 0
    });

    setTitle("");
    setAuthor("");
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={title} onChange={(e) => setTitle(e.target.value)} />
      <input value={author} onChange={(e) => setAuthor(e.target.value)} />
      <button>Add Post</button>
    </form>
  );
}
3) ❤️ PostCard
function PostCard({ post, onLike, onDelete, onSelect }) {
  return (
    <div>
      <h3 onClick={() => onSelect(post)}>{post.title}</h3>
      <p>{post.author}</p>
      <p>Likes: {post.likes}</p>

      <button onClick={() => onLike(post.id)}>Like</button>
      <button onClick={() => onDelete(post.id)}>Delete</button>
    </div>
  );
}
4) 📋 PostList
function PostList({ posts, onLike, onDelete, onSelect }) {
  return (
    <div>
      {posts.map((post) => (
        <PostCard
          key={post.id}
          post={post}
          onLike={onLike}
          onDelete={onDelete}
          onSelect={onSelect}
        />
      ))}
    </div>
  );
}
5) 👀 Selected Preview
function SelectedPostPreview({ post }) {
  if (!post) return <p>Select a post</p>;

  return (
    <div>
      <h2>{post.title}</h2>
      <p>Author: {post.author}</p>
      <p>Likes: {post.likes}</p>
    </div>
  );
}
6) 🧠 App Logic
➕ add
function handleAdd(post) {
  setPosts((prev) => [...prev, post]);
}
❤️ like
function handleLike(id) {
  setPosts((prev) =>
    prev.map((post) =>
      post.id === id
        ? { ...post, likes: post.likes + 1 }
        : post
    )
  );
}
❌ delete
function handleDelete(id) {
  setPosts((prev) => prev.filter((post) => post.id !== id));
}
